def check_number(n):
    # Check if positive, negative, or zero
    if n > 0:
        print("The number is positive.")
    elif n < 0:
        print("The number is negative.")
    else:
        print("The number is zero.")

    # Check if even or odd
    if n % 2 == 0:
        print("The number is even.")
    else:
        print("The number is odd.")
